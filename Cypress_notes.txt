Cypress.Commands.add('addTodo', (text) => {
  cy.get('[data-cy="todo-input"]').type(text)
  cy.get('[data-cy="add-button"]').click()
})

Cypress.Commands.add('completeTodo', (index) => {
  cy.get('[data-cy="todo-item"]')
    .eq(index)
    .find('[data-cy="todo-checkbox"]')
    .check()
})

it('manages todos', () => {
  cy.visit('/')
  cy.addTodo('Buy groceries')
  cy.addTodo('Walk dog')
  cy.completeTodo(0)
})

Mocking Network Requests
Intercept and stub a request
// Match By URL Pattern
cy.intercept('/api/todos')              // Exact match
cy.intercept('/api/todos/*')            // Wildcard
cy.intercept('**/todos')                // Any domain
cy.intercept(/todos/)                   // Regex

// Match By Method
cy.intercept('GET', '/api/todos')
cy.intercept('POST', '/api/todos')
cy.intercept('DELETE', '/api/todos/*')

// Match By Both
cy.intercept({ method: 'GET', url: '/api/todos' })

// Give Requests a Name to Wait for them
cy.intercept('GET', '/api/todos').as('getTodos')

cy.visit('/')

// Wait for the request to complete
cy.wait('@getTodos')

cy.get('[data-cy="todo-item"]').should('have.length', 2)
